// Atomic checkout transaction
prisma.$transaction(async (tx) => {
  // 1. Validate cart & stock
  // 2. Create Order record
  // 3. Create OrderItems (snapshot price + name)
  // 4. Decrement product stock
  // 5. Mark cart as CHECKED_OUT
  // 6. Commit — all or nothing
})

Orders

Order Status Flow

        

          PLACED
          →
          CONFIRMED
          →
          SHIPPED
          →
          DELIVERED
          |
          CANCELLED
        

      


      

        

          POST
          /orders/checkout
          Convert cart to order
          ▼
        

        

          🔑 JWT Required
          

            

              
No Body Required

              
// Uses the authenticated user's
// active cart automatically.
// Cart must have items &
// all items must be in stock.

            

            

              
Response 201

              
{
  "order": {
    "id": "order-uuid",
    "status": "PLACED",
    "totalAmount": 3598,
